\section{Risk factors and risk mapping}
Given a financial position with value $V_t$ at time $t$, determine the change of its value $\Delta V$ over a time horizon $(0,t)$.

\begin{enumerate}
  \item Express $V$ as a function (\textit{risk mapping}) of a set of \textit{risk factors} $\mathbf{Y_t}=(Y^1_t, \dots, Y^N_t)^T$.\\
    f: $\mathbf{Y}_t \mapsto V_t$
  \item Determine the change in value $\Delta V$ as a function of a factor changes\\
    \begin{sit}[Exact]
    \item $\Delta V = f(\mathbf{Y}_t) - f(\mathbf{Y}_0)$
    \item Absolute change: Substituting $\mathbf{Y}_t = \mathbf{Y}_0 + \Delta \mathbf{Y}$ yields\\
  $\Delta V = f(\mathbf{Y}_0 + \Delta \mathbf{Y}) - f(\mathbf{Y}_0)$
    \end{sit}
% 
    \begin{sit}[Approximation ($n$-th order Taylor)]
    \item $\Delta V \approx \frac{\partial}{\partial \mathbf{Y}_0} f(\mathbf{Y}_0) \cdot \Delta \mathbf{Y}$
    \item Relative change: Substituting $\Delta \mathbf{Y} = \mathbf{Y}_0 \circ R(\mathbf{Y})$ yields\\
      $\Delta V \approx \left(\mathbf{Y}_0 \circ \frac{\partial}{\partial \mathbf{Y}_0} f(\mathbf{Y}_0)\right) \cdot R(\mathbf{Y})$ \\
      $R(\mathbf{Y}) = (\Delta Y^1/Y^1_0, \dots, \Delta Y^N_0/Y^N_0)^T$
    \item A \textit{Delta Equivalent} $Y_0 \frac{\partial}{\partial y_i}f(\mathbf{Y}_0)$ maps (as a linear factor) a change in a given risk factor to a change in $V$
    \end{sit}
\end{enumerate}
